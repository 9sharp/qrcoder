/**
 * Encode an image to QR code.
 * Base64 String, JPEG formated.
 *
 * @param {imageStr} image to be encoded
 * @return {qrStr} encoded image
 */

qrcoder.encodeQR = function(imageStr) {
    var qrStr = qr.image({
        size: 8,
        level: 'M',
        value: imageStr
    }).src;
    return qrStr;
};
